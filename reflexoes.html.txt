// Render
function render(filter) {
  const main = document.getElementById('main');
  main.innerHTML = '';

  books.forEach(book => {
    const section = document.createElement('div');
    section.className = `book-section book-${book.id}`;
    if (filter !== 'all' && filter !== book.id) {
      section.classList.add('hidden');
    }

    section.innerHTML = `
      <div class="book-header">
        <span class="book-dot"></span>
        <span class="book-title">${book.title}</span>
        <span class="book-meta">${book.author} · ${book.date}</span>
      </div>
      <div class="cards-grid">
        ${book.insights.map((ins, i) => `
          <div class="card" data-book="${book.id}" data-index="${i}">
            <div class="card-num">${ins.num}</div>
            <span class="card-emoji">${ins.emoji}</span>
            <div class="card-title">${ins.title}</div>
            ${ins.quote ? `<div class="card-quote">"${ins.quote}"</div>` : ''}
            <div class="card-lesson">${ins.lesson}</div>
            <div class="card-expand-hint">Clique para ver mais →</div>
          </div>
        `).join('')}
      </div>
    `;

    main.appendChild(section);
  });

  // card click listeners
  document.querySelectorAll('.card').forEach(card => {
    card.addEventListener('click', () => {
      const bookId = card.dataset.book;
      const idx = parseInt(card.dataset.index);
      openModal(bookId, idx);
    });
  });
}

// Modal
function openModal(bookId, idx) {
  const book = books.find(b => b.id === bookId);
  const ins = book.insights[idx];

  document.getElementById('modal-tag').textContent = book.title;
  document.getElementById('modal-tag').style.background = book.color;
  document.getElementById('modal-num').textContent = `Insight ${ins.num}`;
  document.getElementById('modal-emoji').textContent = ins.emoji;
  document.getElementById('modal-title').textContent = ins.title;
  document.getElementById('modal-lesson').textContent = ins.lesson;
  document.getElementById('modal-action').textContent = ins.action;

  const quoteSection = document.getElementById('modal-quote-section');
  const quoteEl = document.getElementById('modal-quote');
  if (ins.quote) {
    quoteEl.textContent = `"${ins.quote}"`;
    quoteEl.style.borderLeftColor = book.color;
    quoteSection.style.display = 'block';
  } else {
    quoteSection.style.display = 'none';
  }

  document.getElementById('modal-overlay').classList.add('open');
}

document.getElementById('modal-close').addEventListener('click', () => {
  document.getElementById('modal-overlay').classList.remove('open');
});

document.getElementById('modal-overlay').addEventListener('click', (e) => {
  if (e.target === document.getElementById('modal-overlay')) {
    document.getElementById('modal-overlay').classList.remove('open');
  }
});

document.addEventListener('keydown', (e) => {
  if (e.key === 'Escape') document.getElementById('modal-overlay').classList.remove('open');
});

// Filters
let active = 'all';
document.querySelectorAll('.filter-btn').forEach(btn => {
  btn.addEventListener('click', () => {
    document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
    btn.classList.add('active');
    active = btn.dataset.book;
    render(active);
  });
});

// Init
render('all');
